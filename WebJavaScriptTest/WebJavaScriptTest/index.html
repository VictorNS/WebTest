var callback = (function () {
	var i = 0;
	return function (a) {
		i++;
		return a + i;
	};
})(0);
var str = 'Jack Jack Jack'.replace(/Jack/g, callback);
console.log(str); // Jack1 Jack2 Jack3

function add(a) {
	var f = function (b) {
		return a + b;
	};
	return f;
}
console.log(add(5)(7)); // 12
